## 26
create table FILIALES(
FILNUM int(11),
FILVILLE varchar(255)
) engine = InnoDB;

alter table FILIALES
add constraint primary key pk_filiales (FILNUM, FILVILLE);

alter table FILIALES
add constraint fk_filiales_fournisseur foreign key (FILNUM)
references FOURNISSEUR(FRSNUM) on delete cascade;
